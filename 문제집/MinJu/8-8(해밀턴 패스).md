#include <string>
#include <vector>
using namespace std;

bool visited[50] = {false};

class HamiltonPath{
public:
	vector <string> roads;

	int countPaths(vector <string> roads){
		int group = 0, free = 0;
		int connect[50] = {0};

		long long sum = 1;

		this->roads = roads;

		for(int i=0;i<roads.size();i++){
			int yNums = 0;
			for(int j=0;j<roads[i].size();j++){
				if(roads[i][j] == 'Y'){
					yNums++;
					if(yNums > 2)
						return 0;
				}
				connect[i] = yNums;
			}
		}
		
		for(int i=0;i<roads.size();i++){
			if(connect[i] == 0){
				visited[i] = true;
				free++;
			}
			else if(connect[i] == 1 && !visited[i]){
				group++;
				dfs(i);
			}
		}
		
		for(int i=0;i<roads.size();i++)
			if(!visited[i])
				return 0;
		for(int i=0;i<group+free;i++)
			sum = sum * (i+1)%1000000007;
		for(int i=0;i<group;i++)
			sum = sum * 2 % 1000000007;
		return (int)sum;
	}

	void dfs(int city){
		visited[city] = true;
		for(int i=0;i<roads[city].size();i++){
			if(roads[city][i] == 'Y' && !visited[i])
				dfs(i);
		}
	}
};
